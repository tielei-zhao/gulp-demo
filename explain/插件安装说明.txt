
1. 自动加载插件
	cnpm install --save-dev gulp-load-plugins
2. 重命名
	cnpm install --save-dev gulp-rename
3. js文件压缩
	cnpm install --save-dev gulp-uglify
4. css文件压缩
	cnpm install --save-dev gulp-minify-css
5. html文件压缩
	cnpm install --save-dev gulp-minify-html
6. js代码检查	(这里和网页的不同。因版本升级进行了调整)
	cnpm install --save-dev jshint gulp-jshint  
7. 文件合并
	cnpm install --save-dev gulp-concat
8. less和sass的编译
	cnpm install --save-dev gulp-less
	cnpm install --save-dev gulp-sass
9. 图片压缩
	cnpm install --save-dev gulp-imagemin
10.自动刷新
	cnpm install --save-dev gulp-livereload
11.图片压缩缓存	(避免重复压缩图片)
	cnpm install gulp-cache --save-dev
